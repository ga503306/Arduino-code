#define in1 8
#define in2 9
#define in3 14
#define in4 15
#define Trig 12
#define Echo 13


#include <SoftwareSerial.h>
SoftwareSerial BTSerial(10,11); // RX | TX


void setup(){

  pinMode(in1,OUTPUT);
  pinMode(in2,OUTPUT);
  pinMode(in3,OUTPUT);
  pinMode(in4,OUTPUT);
  pinMode (Trig, OUTPUT);
  pinMode (Echo, INPUT);
  Serial.begin(9600);
Serial.println("Enter AT commands:");
BTSerial.begin(9600);
}

void cargogo(){
  digitalWrite(in1,HIGH);
  digitalWrite(in2,LOW);
  digitalWrite(in3,HIGH);
  digitalWrite(in4,LOW);
   delay(10);
}

void turnright(){
  digitalWrite(in1,LOW);
  digitalWrite(in2,HIGH);
  digitalWrite(in3,HIGH);
  digitalWrite(in4,LOW);
   delay(10);
}

void turnleft(){
  digitalWrite(in1,HIGH);
  digitalWrite(in2,LOW);
  digitalWrite(in3,LOW);
  digitalWrite(in4,HIGH);
   delay(10);
}

void carback(){
  digitalWrite(in1,LOW);
  digitalWrite(in2,HIGH);
  digitalWrite(in3,LOW);
  digitalWrite(in4,HIGH);
   delay(10);
}

void sstop(){
  digitalWrite(in1,LOW);
  digitalWrite(in2,LOW);
  digitalWrite(in3,LOW);
  digitalWrite(in4,LOW);
   delay(10);
}
